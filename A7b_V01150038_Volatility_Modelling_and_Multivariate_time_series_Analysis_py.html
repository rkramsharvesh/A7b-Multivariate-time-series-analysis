import yfinance as yf
import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
from statsmodels.stats.diagnostic import het_arch
from arch import arch_model

# 1. Download 5 years of daily data
ticker = 'JPM'
data = yf.download(ticker, start="2020-01-01", end="2025-07-17")

# Check what columns are available
print("\nColumns in downloaded data:", data.columns)

# 2. Use 'Adj Close' if available, else fallback to 'Close'
if 'Adj Close' in data.columns:
    price = data['Adj Close']
elif 'Close' in data.columns:
    print("Warning: 'Adj Close' not found, using 'Close' instead.")
    price = data['Close']
else:
    raise KeyError("Neither 'Adj Close' nor 'Close' column found in the data.")

price = price.dropna()

# 3. Compute log returns
returns = 100 * np.log(price / price.shift(1))
returns.dropna(inplace=True)
returns.name = 'Log Returns'

# 4. Plot returns
plt.figure(figsize=(10, 4))
returns.plot(title='Daily Log Returns: JPM (20202025)')
plt.ylabel('Log Return (%)')
plt.grid(True)
plt.show()

# 5. ARCH LM Test
lm_stat, lm_pvalue, _, _ = het_arch(returns)
print(f"ARCH LM Test:\nLM Statistic = {lm_stat:.4f}, p-value = {lm_pvalue:.4f}")

# 6. Fit GARCH(1,1)
model = arch_model(returns, vol='Garch', p=1, q=1, dist='normal')
fit = model.fit(disp='off')
print(fit.summary())

# 7. Forecast 3-month (63 trading days) volatility
forecast_horizon = 63
forecast = fit.forecast(horizon=forecast_horizon)
vol_forecast = np.sqrt(forecast.variance.values[-1, :])  # daily volatility (%)

# 8. Plot forecast
plt.figure(figsize=(10, 4))
plt.plot(vol_forecast, marker='o')
plt.title("Forecasted Daily Volatility (Next 63 Days)")
plt.xlabel("Day")
plt.ylabel("Volatility (%)")
plt.grid(True)
plt.show()

# 9. Print average volatility forecast
print(f"\nAverage forecasted daily volatility: {vol_forecast.mean():.2f}%")

## ADF Test ##
from statsmodels.tsa.stattools import adfuller
import numpy as np

# Filter dataset from 2010 onward
df_model = df.loc['2010-01':].copy()

# ADF test function
def run_adf(series, name, diff=False):
    adf_result = adfuller(series.dropna(), autolag='AIC')
    print(f"{'First Difference' if diff else 'Level'} ADF Test for {name}")
    print(f"ADF Statistic: {adf_result[0]:.4f}")
    print(f"p-value: {adf_result[1]:.4f}")
    print(f"Critical Values: {adf_result[4]}")
    print("-" * 50)

# Run ADF test on levels
for col in df_model.columns:
    run_adf(df_model[col], col)

# Log first differences
df_log_diff = np.log(df_model).diff().dropna()

# Run ADF test on first differences
for col in df_log_diff.columns:
    run_adf(df_log_diff[col], col, diff=True)
